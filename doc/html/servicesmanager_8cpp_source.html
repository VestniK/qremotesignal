185 void ServicesManager::receive(const QByteArray& msg)
186 {
187  if ( !d->mSerializer ) {
188  return;
189  }
190  MessageAP message;
191  try {
192  message = d->mSerializer->deserialize(msg);
193  } catch (const MessageParsingException& e) {
194  Message err;
195  err.setType(Message::Error);
196  err.setErrorType(e.mErrorType);
197  err.setError( e.reason() );
198  emit send( d->mSerializer->serialize(err) );
199  emit clientError(this, err.errorType(), err.error());
200  return;
201  }
202  if ( message->type() == Message::Error ) {
203  emit error(this, message->errorType(), message->error());
204  return;
205  }
206  QMap<QString,AbsService*>::iterator res = d->mServices.find(message->service());
207  if ( res != d->mServices.end() ) {
208  try {
209  (*res)->processMessage(*message);
210  } catch ( IncorrectMethodException& e ) {
211  Message err;
212  err.setType(Message::Error);
213  err.setErrorType(Message::IncorrectMethod);
214  err.setError(e.reason());
215  err.setService(message->service());
216  err.setMethod(message->method());
217  emit send( d->mSerializer->serialize(err) );
218  emit clientError(this, err.errorType(), err.error());
219  return;
220  }
221  } else {
222  Message err;
223  err.setType(Message::Error);
224  err.setErrorType(Message::UnknownService);
225  err.setError(QString("Unknown service: \"%1\"").arg(message->service()));
226  err.setService(message->service());
227  emit send( d->mSerializer->serialize(err) );
228  emit clientError(this, err.errorType(), err.error());
229  return;
230  }
231 }

333 void ServicesManager::addDevice(QIODevice *dev)
334 {
335  foreach (const QSharedPointer<internals::DeviceManager> dm, d->mDevManagers) {
336  if ( dm->device() == dev ) {
337  return;
338  }
339  }
340  QSharedPointer<internals::DeviceManager> dm(new internals::DeviceManager());
341  dm->setMaxMessageSize(d->mMessageSizeLimit);
342  connect( dm.data(), SIGNAL(received(QByteArray)),
343  this, SLOT(receive(const QByteArray&)) );
344  connect( this, SIGNAL(send(QByteArray)),
345  dm.data(), SLOT(send(const QByteArray&)) );
346  connect( dm.data(), SIGNAL(messageTooBig(qrs::internals::DeviceManager *)),
347  this, SLOT(onMessageTooBig(qrs::internals::DeviceManager *)) );
348  dm->setDevice(dev);
349  d->mDevManagers.append(dm);
350  connect( dev, SIGNAL(destroyed( QObject* )),
351  this, SLOT(onDeviceDeleted(QObject*)) );
352 }
